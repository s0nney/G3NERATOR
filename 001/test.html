    private:
        char* data;
        void (*callback)(void*);

    public:
        VulnerableObject() {
            data = new char[100];
            callback = NULL;
        }

        void setCallback(void (*cb)(void*)) {
            callback = cb;
        }

        void processData() {
            if (callback) {
                callback(data);
            }
        }

        ~VulnerableObject() {
            delete[] data;
        }
